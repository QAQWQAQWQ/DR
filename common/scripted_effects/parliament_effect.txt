


new_nat_tre_add = {
	set_temp_variable = {
		temp_old = new_nat_tre
	}
	#旧变量=旧席位数量X
	add_to_variable = {
		var =  new_nat_tre
		value =  temp_new_nat_tre
	}
	#旧席位数量X（10）+要加的临时旧席位数量X（0）
	set_temp_variable = {
		temp_new = temp_old
	}
	#新=旧
	add_to_temp_variable = {
		temp_new = temp_new_nat_tre
	}
	#新变量+要加的临时旧席位数量X（5）
	custom_effect_tooltip = new_nat_tre_tip

	clamp_variable = {
		var = new_nat_tre
		min = 0
		max = 10000
	}
}
